<script>
document.addEventListener('DOMContentLoaded', function () {
    const mainSearch = document.getElementById('mainSearch');
    const promptsGrid = document.getElementById('promptsGrid');
    const noResults = document.getElementById('noResults');
    const filterChips = document.querySelectorAll('.filter-chip:not(.more-btn)');
    const showMoreBtn = document.getElementById('showMore');
    const additionalFilters = document.getElementById('additionalFilters');
    const prompts = document.querySelectorAll('.prompt-card');

    let currentFilter = 'all';
    let currentSearch = '';

    // Search
    if (mainSearch) {
        mainSearch.addEventListener('input', function () {
            currentSearch = this.value.toLowerCase().trim();
            filterPrompts();
            if (currentSearch.length > 1) {
                const browseSection = document.getElementById('browse-prompts');
                if (browseSection) {
                    browseSection.scrollIntoView({ behavior: 'smooth' });
                }
            }
        });
    }

    // Category filter chips
    filterChips.forEach(function (chip) {
        chip.addEventListener('click', function () {
            filterChips.forEach(function (c) { c.classList.remove('active'); });
            this.classList.add('active');
            currentFilter = this.dataset.category;
            filterPrompts();
        });
    });

    // Show/hide extra filters
    if (showMoreBtn) {
        showMoreBtn.addEventListener('click', function () {
            const isVisible = additionalFilters.style.display === 'flex';
            additionalFilters.style.display = isVisible ? 'none' : 'flex';
            this.innerHTML = isVisible
                ? 'More <i class="fas fa-chevron-down"></i>'
                : 'Less <i class="fas fa-chevron-up"></i>';
        });
    }

    function filterPrompts() {
        var visibleCount = 0;
        prompts.forEach(function (prompt) {
            var title = prompt.dataset.title || '';
            var description = prompt.dataset.description || '';
            var category = prompt.dataset.category || '';

            var matchesSearch = !currentSearch ||
                title.indexOf(currentSearch) !== -1 ||
                description.indexOf(currentSearch) !== -1 ||
                category.indexOf(currentSearch) !== -1;

            var matchesFilter = currentFilter === 'all' ||
                category === currentFilter ||
                category.indexOf(currentFilter) !== -1;

            if (matchesSearch && matchesFilter) {
                prompt.style.display = 'flex';
                visibleCount++;
            } else {
                prompt.style.display = 'none';
            }
        });

        if (noResults) noResults.style.display = visibleCount === 0 ? 'block' : 'none';
        if (promptsGrid) promptsGrid.style.display = visibleCount === 0 ? 'none' : 'grid';
    }

    // Copy button handler
    document.addEventListener('click', async function (e) {
        var copyBtn = e.target.closest('.copy-btn, .copy-btn-small');
        if (copyBtn) {
            var text = copyBtn.dataset.clipboardText;
            if (!text) return;
            try {
                await navigator.clipboard.writeText(text);
                var originalHTML = copyBtn.innerHTML;
                copyBtn.innerHTML = '<i class="fas fa-check"></i>';
                copyBtn.classList.add('copied');
                setTimeout(function () {
                    copyBtn.innerHTML = originalHTML;
                    copyBtn.classList.remove('copied');
                }, 2000);
            } catch (err) {
                console.error('Copy failed:', err);
            }
        }
    });

    // Smooth scroll for anchor links
    document.querySelectorAll('a[href^="#"]').forEach(function (anchor) {
        anchor.addEventListener('click', function (e) {
            var target = document.querySelector(this.getAttribute('href'));
            if (target) {
                e.preventDefault();
                target.scrollIntoView({ behavior: 'smooth', block: 'start' });
            }
        });
    });
});
</script>
